/* SQLEditor (SQLite)*/

CREATE TABLE treatment
(
id NUMERIC UNIQUE,
diagnosis TEXT,
pathologies TEXT,
medications TEXT,
start_date TEXT,
patient NUMERIC REFERENCES person (id)
);

CREATE TABLE attendance
(
id NUMERIC PRIMARY KEY,
treatment NUMERIC REFERENCES treatment (id),
data TEXT,
pa_start TEXT,
spo2_start TEXT,
bpm_start TEXT,
pa_end TEXT,
spo_end TEXT,
bpm_end TEXT,
owner NUMERIC REFERENCES person (id),
notes TEXT
);

CREATE TABLE type_person
(
id NUMERIC PRIMARY KEY,
type_person TEXT
);

CREATE TABLE person
(
id NUMERIC PRIMARY KEY,
name TEXT,
born TEXT,
address TEXT,
phone NUMERIC,
mobile NUMERIC,
other_phone NUMERIC,
email TEXT,
doc_type TEXT,
doc_id TEXT,
type_person NUMERIC REFERENCES type_person (id),
health_care TEXT,
notes TEXT
);
